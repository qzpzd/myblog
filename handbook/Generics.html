    # PCA降维

    def pca(X):
    # normalize the features
    X = (X - X.mean()) / X.std()

    # compute the covariance matrix
    X = np.matrix(X)
    cov = (X.T * X) / X.shape[0]

    # perform SVD
    U, S, V = np.linalg.svd(cov)

    return U, S, V
#恢复数据，实际上并不上完全恢复，只是主要对轴线恢复
#这也是进行降维的作用，相当于提取主要特征
def project_data(X, U, k):
    U_reduced = U[:,:k]
    return np.dot(X, U_reduced)
